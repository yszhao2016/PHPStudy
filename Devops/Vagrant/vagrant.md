Vagrant常用命令   

    vagrant box add "boxIdentity" remoteUrlorLocalFile
    
    vagrant init "boxIdentity"                          //初始化box
    
    vagrant init centos/7    //网上下载
    
    vagrant init centos/7  [url|路径]
    
    vagrant up  //没有就下载  创建vm
    
    vagrant ssh
    
    vagrant status
    
    vagrant halt  //关机
    
    
    vagrant destory //删除机器
